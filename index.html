        // Инициализация игры
        let currentQuestion = null;
        let usedQuestions = new Set();
        function initGame() {
            const gameBoard = document.getElementById('gameBoard');
            gameBoard.innerHTML = '';
            
            gameData.forEach((category, catIndex) => {
                const categoryElement = document.createElement('div');
                categoryElement.className = 'category';
                
                const titleElement = document.createElement('div');
                titleElement.className = 'category-title';
                titleElement.innerHTML = `${category.category}<br><small>${category.subtitle}</small>`;
                titleElement.style.color = category.color;
                categoryElement.appendChild(titleElement);
                
                category.questions.forEach(question => {
                    const questionElement = document.createElement('div');
                    questionElement.className = 'question-value';
                    questionElement.textContent = question.value;
                    questionElement.dataset.category = catIndex;
                    questionElement.dataset.question = question.value;
                    
                    // Проверяем, был ли вопрос уже использован
                    const questionKey = `${catIndex}-${question.value}`;
                    if (usedQuestions.has(questionKey)) {
                        questionElement.classList.add('used');
                    } else {
                        questionElement.addEventListener('click', () => showQuestion(catIndex, question.value));
                    }
                    
                    categoryElement.appendChild(questionElement);
                });
                
                gameBoard.appendChild(categoryElement);
            });
        }
        // Показать вопрос
        function showQuestion(categoryIndex, value) {
            const category = gameData[categoryIndex];
            const question = category.questions.find(q => q.value === value);
            
            if (!question) return;
            
            currentQuestion = { categoryIndex, value };
            
            document.getElementById('modalCategory').innerHTML = 
                `<i class="fas fa-star" style="color:${category.color};"></i> ${category.category}: ${value} очков <i class="fas fa-star" style="color:${category.color};"></i>`;
            document.getElementById('questionText').textContent = question.question;
            document.getElementById('answerText').textContent = `Ответ: ${question.answer}`;
            document.getElementById('answerText').style.display = 'none';
            document.getElementById('showAnswerBtn').style.display = 'block';
            document.getElementById('nextBtn').style.display = 'none';
            
            document.getElementById('questionModal').style.display = 'flex';
            
            // Помечаем вопрос как использованный
            const questionKey = `${categoryIndex}-${value}`;
            usedQuestions.add(questionKey);
        }
        // Закрыть модальное окно
